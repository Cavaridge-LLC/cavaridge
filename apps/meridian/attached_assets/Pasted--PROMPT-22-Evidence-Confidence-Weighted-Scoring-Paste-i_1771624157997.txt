# PROMPT 22: Evidence-Confidence Weighted Scoring

> **Paste into Replit Agent.** Fixes the scoring system so pillars without
> evidence don't get a fake 5.0. Introduces an evidence confidence factor
> that caps scores based on how much documentation supports the assessment.
> A pillar with zero documents = "Not Assessed" at 3.0, not a perfect score.

Replace the current pillar scoring function with:

function calculatePillarScore(pillarId, dealId) {
  // Step 1: Count findings by severity (only non-remediated)
  const findings = getFindings(pillarId, dealId, status != 'remediated');
  const critical = findings.filter(f => f.severity === 'critical').length;
  const high = findings.filter(f => f.severity === 'high').length;
  const medium = findings.filter(f => f.severity === 'medium').length;
  const low = findings.filter(f => f.severity === 'low').length;

  // Step 2: Calculate finding-based score (same as before)
  let findingScore = 5.0;
  findingScore -= critical * 0.8;
  findingScore -= high * 0.5;
  findingScore -= medium * 0.25;
  findingScore -= low * 0.1;
  findingScore = Math.max(1.0, findingScore);  // Floor at 1.0

  // Step 3: Count documents supporting this pillar
  const docCount = countDocumentsForPillar(pillarId, dealId);
  
  // Step 4: Determine evidence confidence tier
  let confidence, scoreCap, confidenceLabel;
  if (docCount === 0) {
    confidence = 0.0;
    scoreCap = 3.0;
    confidenceLabel = 'insufficient';
  } else if (docCount <= 2) {
    confidence = 0.25;
    scoreCap = 4.0;
    confidenceLabel = 'low';
  } else if (docCount <= 5) {
    confidence = 0.65;
    scoreCap = 4.8;
    confidenceLabel = 'moderate';
  } else {
    confidence = 1.0;
    scoreCap = 5.0;
    confidenceLabel = 'high';
  }

  // Step 5: Apply evidence cap
  // If findings brought the score below the cap, use the finding score 
  // (findings can always pull scores DOWN regardless of evidence level)
  // If no findings exist, cap at the evidence tier's maximum
  const finalScore = Math.min(findingScore, scoreCap);

  return {
    score: Math.round(finalScore * 100) / 100,  // 2 decimal places
    finding_score: findingScore,
    evidence_confidence: confidence,
    confidence_label: confidenceLabel,
    score_cap: scoreCap,
    document_count: docCount,
    finding_counts: { critical, high, medium, low }
  };
}

KEY PRINCIPLE: Evidence confidence can only LIMIT the upside. Findings 
always pull scores down. So:
  - 0 documents, 0 findings → 3.0 (capped, not assessed)
  - 0 documents, 2 critical findings → 3.0 (findings pulled to 3.4, 
    but evidence cap is 3.0 so cap applies... WAIT — findings should 
    still count even without evidence docs. If someone manually added 
    findings, those are real.)

CORRECTION — refined logic:
  - Finding score ALWAYS applies (findings are evidence themselves)
  - Evidence cap ONLY applies to the "no findings" baseline
  - If finding_score < score_cap → use finding_score (findings win)
  - If finding_score >= score_cap → use score_cap (cap the upside)

So the formula is:
  final_score = Math.min(finding_score, score_cap)

This means:
  - 0 docs, 0 findings → min(5.0, 3.0) = 3.0 ← correct, "not assessed"
  - 0 docs, 1 critical → min(4.2, 3.0) = 3.0 ← findings + no evidence
  - 2 docs, 0 findings → min(5.0, 4.0) = 4.0 ← limited evidence, looks ok
  - 2 docs, 1 critical → min(4.2, 4.0) = 4.0 ← cap still applies
  - 6 docs, 0 findings → min(5.0, 5.0) = 5.0 ← well documented, truly clean
  - 6 docs, 3 critical → min(2.6, 5.0) = 2.6 ← findings drive score down

function calculateCompositeScore(dealId) {
  const pillars = getAllPillars(dealId);
  
  // Weighted average of pillar scores (same as before)
  let weightedSum = 0;
  let totalWeight = 0;
  pillars.forEach(p => {
    weightedSum += p.score * p.weight;
    totalWeight += p.weight;
  });
  let composite = weightedSum / totalWeight;
  
  // Critical penalty (same as before)
  const criticalCount = getTotalCriticalFindings(dealId);
  const criticalPenalty = Math.min(criticalCount * 0.05, 0.30);
  composite = composite * (1 - criticalPenalty);
  
  // NEW: Overall evidence penalty
  // If MORE THAN HALF of pillars are "insufficient" or "low", apply 
  // an additional confidence penalty
  const insufficientCount = pillars.filter(
    p => p.confidence_label === 'insufficient'
  ).length;
  const lowCount = pillars.filter(
    p => p.confidence_label === 'low'
  ).length;
  
  const weakPillarCount = insufficientCount + lowCount;
  const totalPillars = pillars.length;
  
  let overallConfidence;
  if (insufficientCount === totalPillars) {
    overallConfidence = 'insufficient';
  } else if (weakPillarCount > totalPillars / 2) {
    overallConfidence = 'low';
  } else if (weakPillarCount > 0) {
    overallConfidence = 'moderate';
  } else {
    overallConfidence = 'high';
  }
  
  // Scale to 0-100
  const score100 = (composite / 5) * 100;
  
  return {
    composite_score: Math.round(score100 * 10) / 10,
    overall_confidence: overallConfidence,
    insufficient_pillars: insufficientCount,
    low_pillars: lowCount,
    moderate_pillars: pillars.filter(p => p.confidence_label === 'moderate').length,
    high_pillars: pillars.filter(p => p.confidence_label === 'high').length
  };
}
