async function resolveFilePath(doc) {
  const fs = require('fs');
  const path = require('path');
  
  // Try the stored path as-is
  if (doc.file_path && fs.existsSync(doc.file_path)) {
    return doc.file_path;
  }
  
  // Try common variations
  const candidates = [
    doc.file_path,
    path.join(process.cwd(), doc.file_path),
    path.join(process.cwd(), 'uploads', doc.filename),
    path.join(process.cwd(), 'data', 'uploads', doc.filename),
    path.join(process.cwd(), 'public', 'uploads', doc.filename),
    path.join(process.cwd(), 'uploads', doc.id + path.extname(doc.filename)),
    path.join(process.cwd(), 'data', 'uploads', doc.id + path.extname(doc.filename)),
    // For ZIP-extracted files, try the parent folder
    doc.file_path ? path.join(process.cwd(), doc.file_path.replace(/^\//, '')) : null,
  ].filter(Boolean);
  
  for (const candidate of candidates) {
    if (fs.existsSync(candidate)) {
      console.log(`File found at: ${candidate}`);
      return candidate;
    }
  }
  
  console.error(`File not found for doc ${doc.id} (${doc.filename}). Tried:`, candidates);
  return null;
}

const { Client } = require('@replit/object-storage');
const storage = new Client();

async function getFileFromStorage(doc) {
  try {
    // Try object storage key patterns
    const keys = [
      doc.file_path,
      `uploads/${doc.filename}`,
      `uploads/${doc.id}/${doc.filename}`,
      doc.id,
    ].filter(Boolean);
    
    for (const key of keys) {
      try {
        const data = await storage.downloadAsBytes(key);
        if (data) return data;
      } catch (e) {
        // Try next key
      }
    }
    return null;
  } catch (e) {
    console.error('Object storage error:', e);
    return null;
  }
}

async function loadPreview(documentId) {
  try {
    const response = await fetch(`/api/documents/${documentId}/preview`);
    
    if (!response.ok) {
      const error = await response.json();
      console.error('Preview API error:', error);
      setPreviewState('error');
      setErrorInfo(error.debug || {});
      return;
    }
    
    const contentType = response.headers.get('Content-Type');
    
    // Binary content (images, PDFs)
    if (contentType?.startsWith('image/') || contentType === 'application/pdf') {
      const blob = await response.blob();
      const url = URL.createObjectURL(blob);
      setPreviewUrl(url);
      setPreviewType(contentType.startsWith('image/') ? 'image' : 'pdf');
      setPreviewState('loaded');
      return;
    }
    
    // JSON content (text files, extracted text)
    const data = await response.json();
    if (data.content) {
      setTextContent(data.content);
      setLanguage(data.language || 'text');
      setPreviewType('text');
      setPreviewState('loaded');
      return;
    }
    
    // Fallback
    setPreviewState('unavailable');
    
  } catch (error) {
    console.error('Preview load error:', error);
    setPreviewState('error');
  }
}
