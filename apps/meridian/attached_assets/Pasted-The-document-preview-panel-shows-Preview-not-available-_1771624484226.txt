COMMON CAUSES AND FIXES:

1. FILE TYPE MISMATCH:
   The file_type column might store "text/markdown" (MIME type) while 
   the preview component checks for "md" (extension), or vice versa.
   
   Fix: normalize to EXTENSION format everywhere. In the preview component, 
   add a function:
   
   function getFileExtension(doc) {
     // Try the file_type column first
     let ext = doc.file_type?.toLowerCase();
     
     // If it's a MIME type, convert to extension
     const mimeMap = {
       'text/markdown': 'md',
       'text/plain': 'txt',
       'text/html': 'html',
       'text/css': 'css',
       'text/csv': 'csv',
       'application/json': 'json',
       'application/xml': 'xml',
       'application/pdf': 'pdf',
       'application/zip': 'zip',
       'application/vnd.openxmlformats-officedocument.wordprocessingml.document': 'docx',
       'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet': 'xlsx',
       'application/vnd.openxmlformats-officedocument.presentationml.presentation': 'pptx',
       'image/png': 'png',
       'image/jpeg': 'jpg',
       'image/gif': 'gif',
       'image/webp': 'webp',
       'image/svg+xml': 'svg',
       'message/rfc822': 'eml',
       'application/x-yaml': 'yaml',
     };
     
     if (mimeMap[ext]) {
       ext = mimeMap[ext];
     }
     
     // If still no match, extract from filename
     if (!ext || ext.includes('/')) {
       const filename = doc.filename || doc.original_filename || '';
       const parts = filename.split('.');
       if (parts.length > 1) {
         ext = parts[parts.length - 1].toLowerCase();
       }
     }
     
     return ext || 'unknown';
   }

   function getPreviewType(ext) {
     const imageExts = ['png', 'jpg', 'jpeg', 'gif', 'webp', 'tiff', 'svg', 'bmp'];
     const textExts = ['md', 'txt', 'log', 'cfg', 'conf', 'ini', 'env', 
                       'json', 'xml', 'yaml', 'yml', 'csv', 'tsv',
                       'html', 'htm', 'css', 'js', 'jsx', 'ts', 'tsx',
                       'py', 'sh', 'bash', 'sql', 'rb', 'go', 'rs',
                       'java', 'c', 'cpp', 'h', 'ps1', 'bat', 'cmd'];
     const docExts = ['docx', 'doc'];
     const spreadExts = ['xlsx', 'xls'];
     const slideExts = ['pptx', 'ppt'];
     const pdfExts = ['pdf'];
     const emailExts = ['eml', 'msg'];
     
     if (imageExts.includes(ext)) return 'image';
     if (pdfExts.includes(ext)) return 'pdf';
     if (textExts.includes(ext)) return 'text';
     if (docExts.includes(ext)) return 'document';
     if (spreadExts.includes(ext)) return 'spreadsheet';
     if (slideExts.includes(ext)) return 'slides';
     if (emailExts.includes(ext)) return 'email';
     return 'fallback';
   }
