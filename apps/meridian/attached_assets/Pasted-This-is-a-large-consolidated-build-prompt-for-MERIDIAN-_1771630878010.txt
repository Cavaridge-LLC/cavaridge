function calculatePillarScore(pillarId, dealId) {
  // Step 1: Count non-remediated findings by severity
  const findings = await getFindings(pillarId, dealId);
  const critical = findings.filter(f => f.severity === 'critical' && f.status !== 'remediated').length;
  const high = findings.filter(f => f.severity === 'high' && f.status !== 'remediated').length;
  const medium = findings.filter(f => f.severity === 'medium' && f.status !== 'remediated').length;
  const low = findings.filter(f => f.severity === 'low' && f.status !== 'remediated').length;

  // Step 2: Calculate finding-based score WITH diminishing returns
  let findingScore = 5.0;
  findingScore -= critical * 0.8;
  findingScore -= high * 0.5;
  // Medium: full penalty for first 5, reduced after
  findingScore -= Math.min(medium, 5) * 0.25 + Math.max(0, medium - 5) * 0.10;
  // Low: full penalty for first 3, reduced after
  findingScore -= Math.min(low, 3) * 0.1 + Math.max(0, low - 3) * 0.03;

  // Step 3: FLOOR — never below 1.0
  findingScore = Math.max(1.0, findingScore);

  // Step 4: Count documents mapped to this pillar
  const docCount = await countDocumentsForPillar(pillarId, dealId);

  // Step 5: Determine evidence cap
  let scoreCap, confidenceLabel;
  if (docCount === 0) { scoreCap = 3.0; confidenceLabel = 'insufficient'; }
  else if (docCount <= 2) { scoreCap = 4.0; confidenceLabel = 'low'; }
  else if (docCount <= 5) { scoreCap = 4.8; confidenceLabel = 'moderate'; }
  else { scoreCap = 5.0; confidenceLabel = 'high'; }

  // Step 6: APPLY THE CAP — this is the critical fix
  const finalScore = Math.min(findingScore, scoreCap);

  // Step 7: Final safety floor
  const safeScore = Math.max(1.0, finalScore);

  // Step 8: SAVE THE CAPPED SCORE (not findingScore, not the uncapped value)
  await updatePillar(pillarId, {
    score: Math.round(safeScore * 100) / 100,
    score_cap: scoreCap,
    evidence_confidence: docCount === 0 ? 0 : docCount <= 2 ? 0.25 : docCount <= 5 ? 0.65 : 1.0,
    confidence_label: confidenceLabel,
    document_count: docCount,
    finding_count: critical + high + medium + low
  });

  return safeScore;
}
